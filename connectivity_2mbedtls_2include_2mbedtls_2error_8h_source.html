33/**
34 * \defgroup mbedtls_errors Error Codes
35 * \ingroup mbedtls
36 *
37 * Error code layout.
38 *
39 * Currently we try to keep all error codes within the negative space of 16
40 * bits signed integers to support all platforms (-0x0001 - -0x7FFF). In
41 * addition we'd like to give two layers of information on the error if
42 * possible.
43 *
44 * For that purpose the error codes are segmented in the following manner:
45 *
46 * 16 bit error code bit-segmentation
47 *
48 *
49 * \li 1 bit - Unused (sign bit)
50 * \li 3 bits - High level module ID
51 * \li 5 bits - Module-dependent error code
52 * \li 7 bits - Low level module errors
53 *
54 * For historical reasons, low-level error codes are divided in even and odd,
55 * even codes were assigned first, and -1 is reserved for other errors.
56 *
57 * \code{.unparsed}
58 * Low-level module errors (0x0002-0x007E, 0x0001-0x007F)
59 *
60 * Module Nr Codes assigned
61 * ERROR 2 0x006E 0x0001
62 * MPI 7 0x0002-0x0010
63 * GCM 3 0x0012-0x0014 0x0013-0x0013
64 * BLOWFISH 3 0x0016-0x0018 0x0017-0x0017
65 * THREADING 3 0x001A-0x001E
66 * AES 5 0x0020-0x0022 0x0021-0x0025
67 * CAMELLIA 3 0x0024-0x0026 0x0027-0x0027
68 * XTEA 2 0x0028-0x0028 0x0029-0x0029
69 * BASE64 2 0x002A-0x002C
70 * OID 1 0x002E-0x002E 0x000B-0x000B
71 * PADLOCK 1 0x0030-0x0030
72 * DES 2 0x0032-0x0032 0x0033-0x0033
73 * CTR_DBRG 4 0x0034-0x003A
74 * ENTROPY 3 0x003C-0x0040 0x003D-0x003F
75 * NET 13 0x0042-0x0052 0x0043-0x0049
76 * ARIA 4 0x0058-0x005E
77 * ASN1 7 0x0060-0x006C
78 * CMAC 1 0x007A-0x007A
79 * PBKDF2 1 0x007C-0x007C
80 * HMAC_DRBG 4 0x0003-0x0009
81 * CCM 3 0x000D-0x0011
82 * ARC4 1 0x0019-0x0019
83 * MD2 1 0x002B-0x002B
84 * MD4 1 0x002D-0x002D
85 * MD5 1 0x002F-0x002F
86 * RIPEMD160 1 0x0031-0x0031
87 * SHA1 1 0x0035-0x0035 0x0073-0x0073
88 * SHA256 1 0x0037-0x0037 0x0074-0x0074
89 * SHA512 1 0x0039-0x0039 0x0075-0x0075
90 * CHACHA20 3 0x0051-0x0055
91 * POLY1305 3 0x0057-0x005B
92 * CHACHAPOLY 2 0x0054-0x0056
93 * PLATFORM 2 0x0070-0x0072
94 *
95 * High-level module nr (3 bits - 0x0...-0x7...)
96 * Name ID Nr of Errors
97 * PEM 1 9
98 * PKCS#12 1 4 (Started from top)
99 * X509 2 20
100 * PKCS5 2 4 (Started from top)
101 * DHM 3 11
102 * PK 3 15 (Started from top)
103 * RSA 4 11
104 * ECP 4 10 (Started from top)
105 * MD 5 5
106 * HKDF 5 1 (Started from top)
107 * SSL 5 2 (Started from 0x5F00)
108 * CIPHER 6 8 (Started from 0x6080)
109 * SSL 6 24 (Started from top, plus 0x6000)
110 * SSL 7 32
111 *
112 * Module dependent error code (5 bits 0x.00.-0x.F8.)
113 * \endcode
114 */

128/**
129 * \brief Translate a mbed TLS error code into a string representation,
130 * Result is truncated if necessary and always includes a terminating
131 * null byte.
132 *
133 * \param errnum error code
134 * \param buffer buffer to place representation in
135 * \param buflen length of the buffer
136 */
137void mbedtls_strerror( int errnum, char *buffer, size_t buflen );
138
139/**
140 * \brief Translate the high-level part of an Mbed TLS error code into a string
141 * representation.
142 *
143 * This function returns a const pointer to an un-modifiable string. The caller
144 * must not try to modify the string. It is intended to be used mostly for
145 * logging purposes.
146 *
147 * \param error_code error code
148 *
149 * \return The string representation of the error code, or \c NULL if the error
150 * code is unknown.
151 */
152const char * mbedtls_high_level_strerr( int error_code );
153
154/**

160 * logging purposes.
161 *
162 * \param error_code error code
163 *
164 * \return The string representation of the error code, or \c NULL if the error
165 * code is unknown.

167const char * mbedtls_low_level_strerr( int error_code );
