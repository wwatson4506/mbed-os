201/** Registers a handler to receive event callbacks.
202 *
203 * This function registers a callback handler to be notified when
204 * a particular event is received.
205 *
206 *
207 * @note Currently each event may only be registered to one handler and there is a limit to the number of simultaneously
208 * registered events. Maximum of 5 event handlers can registered simultaneously, this also includes the internal
209 * event handler registration which happens during scan, join and starting an AP.
210 *
211 * @param ifp Pointer to handle instance of whd interface
212 * @param event_type Pointer to the event list array
213 * @param handler_func A function pointer to the handler callback
214 * @param handler_user_data A pointer value which will be passed to the event handler function
215 * at the time an event is triggered (NULL is allowed)
216 * @param event_index Entry where the event handler is registered in the list
217 *
218 * @return WHD_SUCCESS or Error code
219 */
220uint32_t whd_wifi_set_event_handler(whd_interface_t ifp, const uint32_t *event_type, whd_event_handler_t handler_func,
221 void *handler_user_data, uint16_t *event_index);
222/** Registers a handler to receive error callbacks.
223 *
224 * This function registers a callback handler to be notified when
225 * a particular event is received.
226 *
227 *
228 * @note Currently each event may only be registered to one handler and there is a limit to the number of simultaneously
229 * registered events. Maximum of 5 event handlers can registered simultaneously, this also includes the internal
230 * event handler registration which happens during scan, join and starting an AP.
231 *
232 * @param ifp Pointer to handle instance of whd interface
233 * @param error_nums Pointer to the event list as WLC_ERR_BUS and WLC_ERR_FW
234 * @param handler_func A function pointer to the handler callback
235 * @param handler_user_data A pointer value which will be passed to the event handler function

237 * @param error_index Entry where the error handler is registered in the list
238 *
239 * @return WHD_SUCCESS or Error code
